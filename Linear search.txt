#include<stdio.h>

int array;
int i;
int n;

int search(int array[], int value)
{
    for(i = 0; i < n; i++)
    {
        if(array[i] == value)
        {
            printf("The position of the value is %d", i+1);
        }
    }
}

int main()
{
    int value;
    printf("Enter the size of the array: ");
    scanf("%d", &n);
    int array[n];

    printf("Enter the value of the array: ");
    for(i = 0; i < n; i++);
    {
        scanf("%d", &array[i]);
    }

    printf("Enter the value want to search: ");
    scanf("%d", &value);

    search(array, value);
}
